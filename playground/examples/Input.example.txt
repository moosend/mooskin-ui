class InputExample extends React.Component {

    constructor(){
        super();
        this.onChange = this.onChange.bind(this);
        this.validate = this.validate.bind(this);
        this.onClipboardButtonClick = this.onClipboardButtonClick.bind(this);

        this.state = {
            status: '',
            message: '',
            value: 'copy value!!',
            valueAsNumber: 5,
            icon: 'search'
        }
    }

    render(){

        personalizationTags=[
                            {
                                label: 'Unsubscribe Link',
                                value: '#unsubscribeLink#'
                            },
                            {
                                label: 'Update Profile Link',
                                value: '#updateProfileLink#'
                            },
                            {
                                label: 'Forward to a Friend Link',
                                value: '#forwardToFriendLink#'
                            },
                            {
                                label: 'Double Opt-in Verification Link',
                                value: '#VerificationLink#'
                            },
                            {
                                label: 'Account VAT',
                                value: '#account:vat#'
                            },
                            {
                                label: 'Account Email',
                                value: '#account:email#'
                            },
                            {
                                label: 'Account Address',
                                value: '#account:address#'
                            },
                            {
                                label: 'Account Company',
                                value: '#account:company#'
                            },
                            {
                                label: 'Current Date and Time',
                                value: '#datetime#'
                            },
                            {
                                label: 'Recipient Name',
                                value: '#recipient:name#'
                            },
                            {
                                label: 'Recipient Email',
                                value: '#recipient:email#'
                            }];

        return(
            <div>
                <Input icon="check" value={this.state.value} dataLabel='value' type="text" name="olala" onChange={this.onChange} personalizationTags={personalizationTags} emoji type="text" placeholder="max length 5" label="Some label:"/>
                <br/>
                <Input value={this.state.valueAsNumber} minmax={[10, 213]} dataLabel='valueAsNumber' type="number" onChange={this.onChange} label="Number only:"/>
                <br/>
                <Input
                    onChange={this.onChange}
                    value="password"
                    type="password"
                    label="Password:"
                    labelTop
                />
                <br/>
                <Input value={this.state.value} onClipboardButtonClick={this.onClipboardButtonClick} clipboardButton="Copy Value" validate={this.validate} icon={this.state.icon} status={this.state.status} description={this.state.message} onChange={this.onChange} placeholder="placeholder" className=".input-field" autocomplete/>
                <br/>
                <Input onChange={this.onChange} disabled value="disabled" description="This input is disabled"/>
            </div>
        );
    }

    onChange(e, data) {
        console.log(data.value);
        this.setState({[data.dataLabel]: data.value})
    }

    validate(data){
        console.log(data.value);
        if (data.value){
            if (data.value.length < 5){
                this.setState({status: 'error', message: 'Input should have 5 or more characters', icon: 'clear'});
            } else {
                this.setState({status: 'success', message: '', icon: 'check'})
            }
        } else {
            this.setState({status: 'error', message: 'This input field is required', icon: 'clear'});
        }
    }

    onClipboardButtonClick(e, data) {
        alert('Copied: ' + data.value);
    }

}
